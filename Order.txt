import java.util.List;
import java.util.function.DoubleUnaryOperator;
import java.util.function.Supplier;

public class Order {

    private List<Product> products;

    public Order(List<Product> products) {
        this.products = products;
    }

    /**
     * Calculates subtotal using streams
     */
    public double calculateSubtotal() {
        return products.stream()
                .mapToDouble(Product::getPrice)
                .sum();
    }

    /**
     * Applies tax using lambda
     */
    public double applyTax(double taxRate) {
        DoubleUnaryOperator tax = total -> total * taxRate;
        return tax.applyAsDouble(calculateSubtotal());
    }

    /**
     * Calculates shipping using lambda
     */
    public double calculateShipping() {
        Supplier<Double> shipping =
                () -> products.size() > 5 ? 0.0 : 10.0;
        return shipping.get();
    }

    public double calculateTotal(double taxRate) {
        return calculateSubtotal()
                + applyTax(taxRate)
                + calculateShipping();
    }
}
